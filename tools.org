** 10 screenshot.sh

*** my modification

#!/bin/sh
#

screenshot() {
	case $1 in
	full)
		scrot -m
		;;
	window)
		sleep 1
#		scrot -s 
		scrot -s '/tmp/%Y-%m-%d-%k%M_$wx$h.png' -e 'google-chrome-stable $f'
		;;
	*)
		;;
	esac;
}

screenshot $1

echo $f
